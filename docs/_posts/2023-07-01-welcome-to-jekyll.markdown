void flatten_matrix(Matrix tMat, int sRow, int sCol, int* flat_array)
{
  for (int row = 0; row < sRow; ++row) {
    for (int col = 0; col < sCol; ++col)
      flat_array[(row * sCol) + col] = tMat[row][col];
  }
}
